! Revisa las declaraciones mutuamente recursivas

 

let  

 

rec  

 

    proc   a ( )   ~   put ( 'Z' )   end   ! tiene el end

    |  

    func   m ( ) :   Integer   ~   10  

 

end  

 

in  

    putint   ( 1 )  

end
